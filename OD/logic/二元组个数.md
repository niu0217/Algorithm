## 代码

### 自身

```c++
#include<iostream>
#include<vector>
#include<unordered_map>

using namespace std;

int countTuple(vector<int>& numsA, vector<int>& numsB) {
    //key:   元素
    //value: 该元素出现的次数
    unordered_map<int, int> umap;
    for(int i = 0; i < numsA.size(); i++) {
        umap[numsA[i]]++;
    }
    int count = 0;
    for(int i = 0; i < numsB.size(); i++) {
        count += umap[numsB[i]];
    }
    return count;
}

int main()
{
    int value;

    int m;
    cin>>m;
    vector<int> numsA;
    for(int i = 0; i < m; i++) {
        cin>>value;
        numsA.push_back(value);
    }

    int n;
    cin>>n;
    vector<int> numsB;
    for(int i = 0; i < n; i++) {
        cin>>value;
        numsB.push_back(value);
    }

    int result = countTuple(numsA, numsB);
    cout<<result<<endl;
}
```

```c++
#include <iostream>
#include <vector>
#include <unordered_map>

using namespace std;

int count_tuples(vector<int>& a, vector<int>& b) {
    int count = 0;
    unordered_map<int, int> freq; // 用于记录a数组中每个元素的出现频率
    for (int i = 0; i < a.size(); i++) {
        freq[a[i]]++; // 统计a数组中每个元素的出现频率
    }
    for (int i = 0; i < b.size(); i++) {
        count += freq[b[i]]; // 统计b数组中每个元素在a数组中的出现次数，累加到count中
    }
    return count;
}

int main() {
    int m;
    cin >> m; // 输入第一个数组的长度
    vector<int> a(m);
    for (int i = 0; i < m; i++) {
        cin >> a[i]; // 输入第一个数组的元素
    }
    int n;
    cin >> n; // 输入第二个数组的长度
    vector<int> b(n);
    for (int i = 0; i < n; i++) {
        cin >> b[i]; // 输入第二个数组的元素
    }
    int result = count_tuples(a, b); // 调用函数计算二元组个数
    cout << result << endl; // 输出二元组个数
    return 0;
}
```
