# 真正的密码

## [题目链接](https://blog.csdn.net/m0_47384542/article/details/131981710)

## 代码

```c++
#include<iostream>
#include<set>
#include<string>
#include<vector>
#include<sstream>

using namespace std;

int main()
{
    vector<string> strVec;
    string s;
    getline(cin, s);
    istringstream iss(s);
    string word;
    while(iss>>word) {
        strVec.push_back(word);
    }

    set<string> wordSet;
    for(string s : strVec) {
        wordSet.insert(s);
    }

    string truePassward = "";
    for(string s : strVec) {
        //条件1: 检查这个词所有以索引为0开头的子串在数组中是否都有
        bool flag = true;
        for(int i = 1; i < s.size(); i++) {
            string partStr = s.substr(0, i);
            if(wordSet.find(partStr) == wordSet.end()) {
                flag = false;
                break;
            }
        }

        if(flag) {
            //条件2: 比较密码长度
            if(s.size() > truePassward.size()) {
                truePassward = s;
            }
            //条件3: 比较密码字典排序
            else if(s.size() == truePassward.size() && s > truePassward) {
                truePassward = s;
            }
        }
    }

    cout<<truePassward<<endl;
}
```
