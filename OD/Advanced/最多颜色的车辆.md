# 最多颜色的车辆

## [题目描述](https://blog.csdn.net/m0_47384542/article/details/131997471)

## 解题思路

滑动窗口

## 代码

```c++
#include<iostream>
#include<vector>
#include<unordered_map>
#include<sstream>

using namespace std;

int main()
{
    vector<int> carColors;
    int windowSize;
    string s;
    getline(cin, s);
    cin>>windowSize;
    istringstream iss(s);
    int carColor;
    while(iss>>carColor) {
        carColors.push_back(carColor);
    }

    //key:   颜色
    //value: 该颜色出现的次数
    unordered_map<int, int> umap;
    int result = 0;
    for(int i = 0; i < carColors.size(); i++) {
        umap[carColors[i]]++;
        if(i >= windowSize) {
            umap[carColors[i - windowSize]]--;
            if(umap[carColors[i - windowSize]] == 0) {
                umap.erase(carColors[i - windowSize]);
            }
        }
        result = max(result, umap[carColors[i]]);
    }
    cout<<result<<endl;
}
```
