### 自己

```c++
#include<iostream>
#include<vector>

using namespace std;

int main()
{
    int n;
    int m;
    int c;
    int k;
    cin>>n>>m>>c>>k;
    vector<vector<int>> dp(n, vector<int>(m, 0));
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < m; j++) {
            cin>>dp[i][j];
        }
    }
    int result = 0;
    for(int i = 0; i <= n - c; i++) {
        for(int j = 0; j <= m - c; j++) {
            int totalElect = 0;
            for(int x = i; x < i + c; x++) {
                for(int y = j; y < j + c; y++) {
                    totalElect += dp[x][y];
                }
            }
            if(totalElect >= k) {
                result++;
            }
        }
    }
    cout<<result<<endl;
}
```

### 外部

```c++
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n, m, c, k;
    cin >> n >> m >> c >> k;

    // 创建二维向量来存储每个地块的发电量
    vector<vector<int>> land(n, vector<int>(m));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> land[i][j];
        }
    }

    int count = 0;  // 记录满足条件的地块数量

    // 遍历地块的每个可能的正方形子区域
    for (int i = 0; i <= n - c; i++) {
        for (int j = 0; j <= m - c; j++) {
            int total_power = 0;  // 记录子区域的总发电量
            // 计算子区域的总发电量，累加每个地块的发电量
            for (int x = i; x < i + c; x++) {
                for (int y = j; y < j + c; y++) {
                    total_power += land[x][y];
                }
            }
            // 判断总发电量是否满足目标电量
            if (total_power >= k) {
                count++;
            }
        }
    }

    cout << count << endl;  // 输出满足条件的地块数量

    return 0;
}
```
