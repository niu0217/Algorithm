# 停车场车辆统计

## 题目

[8、停车场车辆统计](https://blog.csdn.net/m0_47384542/article/details/131467555)

## 代码

```c++
#include<iostream>
#include<string>
#include<vector>

using namespace std;

int main()
{
    string s;
    cin>>s;
    int slow = 0;
    vector<int> carVec;
    for(int fast = 0; fast < s.size(); fast++) {
        if(s[fast] == ',') {
            string numberStr = s.substr(slow, fast - slow);
            slow = fast + 1;
            carVec.push_back(stoi(numberStr));
        }
    }
    carVec.push_back(stoi(s.substr(slow, s.size() - slow)));

    int result = 0;
    for(int i = 0; i < carVec.size(); i++) {
        if(carVec[i] == 1) {
            if(i + 2 < s.size() && carVec[i + 1] == 1 && carVec[i + 2] == 1) {
                i += 2;
                result++;
            }
            else if(i + 1 < s.size() && carVec[i + 1] == 1) {
                i += 1;
                result++;
            }
            else {
                result++;
            }
        }
    }
    cout<<result<<endl;
}
```
