# 处理器问题

## [题目链接](https://blog.csdn.net/m0_47384542/article/details/131692532)

## 代码

```c++
#include <iostream>
#include <vector>
using namespace std;

int main() {
    vector<int> arr1;
    vector<int> arr2;
    int num;
    string str;

    getline(cin, str);

    // 将处理器编号分别放入arr1和arr2中
    for (int i = 0; i < str.length(); i++) {
        char c = str[i];
        if (c >= '0' && c <= '3') {
            arr1.push_back(c - '0');
        } else if (c >= '4' && c <= '7') {
            arr2.push_back(c - '0');
        }
    }

    cin >> num;

    switch (num) {
        case 1: {
            bool flag = false;
            // 如果申请处理器个数为1，则选择同一链路剩余可用的处理器数量为1个的为最佳
            if (arr1.size() == 1) {
                cout << "[[" << arr1[0] << "]";
                flag = true;
            }
            if (arr2.size() == 1) {
                if (flag) {
                    cout << ", [" << arr2[0] << "]]";
                } else {
                    cout << "[[" << arr2[0] << "]]";
                    flag = true;
                }
            }
            if (flag) {
                cout << endl;
                break;
            }

            // 如果申请处理器个数为1，则选择同一链路剩余可用的处理器数量为3个的为次佳
            if (arr1.size() == 3) {
                cout << "[[" << arr1[0] << "], [" << arr1[1] << "], [" << arr1[2];
                flag = true;
            }
            if (arr2.size() == 3) {
                if (flag) {
                    cout << "], [" << arr2[0] << "], [" << arr2[1] << "], [" << arr2[2];
                } else {
                    cout << "[[" << arr2[0] << "], [" << arr2[1] << "], [" << arr2[2];
                    flag = true;
                }
            }
            if (flag) {
                cout << "]]" << endl;
                break;
            }

            // 如果申请处理器个数为1，则选择同一链路剩余可用的处理器数量为2个的为次次佳
            if (arr1.size() == 2) {
                cout << "[[" << arr1[0] << "], [" << arr1[1];
                flag = true;
            }
            if (arr2.size() == 2) {
                if (flag) {
                    cout << "], [" << arr2[0] << "], [" << arr2[1];
                } else {
                    cout << "[[" << arr2[0] << "], [" << arr2[1];
                    flag = true;
                }
            }
            if (flag) {
                cout << "]]" << endl;
                break;
            }

            // 如果申请处理器个数为1，则选择同一链路剩余可用的处理器数量为4个的为最次佳
            if (arr1.size() == 4) {
                cout << "[[" << arr1[0] << "], [" << arr1[1] << "], [" << arr1[2] << "], [" << arr1[3];
                flag = true;
            }
            if (arr2.size() == 4) {
                if (flag) {
                    cout << "], [" << arr2[0] << "], [" << arr2[1] << "], [" << arr2[2] << "], [" << arr2[3];
                } else {
                    cout << "[[" << arr2[0] << "], [" << arr2[1] << "], [" << arr2[2] << "], [" << arr2[3];
                    flag = true;
                }
            }
            if (flag) {
                cout << "]]" << endl;
            }
            break;
        }

        case 2: {
            bool flag = false;
            // 如果申请处理器个数为2，则选择同一链路剩余可用的处理器数量为2个的为最佳
            if (arr1.size() == 2) {
                cout << "[[" << arr1[0] << ", " << arr1[1];
                flag = true;
            }
            if (arr2.size() == 2) {
                if (flag) {
                    cout << "], [" << arr2[0] << ", " << arr2[1];
                } else {
                    cout << "[[" << arr2[0] << ", " << arr2[1];
                    flag = true;
                }
            }
            if (flag) {
                cout << "]]" << endl;
                break;
            }

            // 如果申请处理器个数为2，则选择同一链路剩余可用的处理器数量为4个的为次佳
            if (arr1.size() == 4) {
                cout << "[[" << arr1[0] << ", " << arr1[1];
                cout << "], [" << arr1[0] << ", " << arr1[2];
                cout << "], [" << arr1[0] << ", " << arr1[3];
                cout << "], [" << arr1[1] << ", " << arr1[2];
                cout << "], [" << arr1[1] << ", " << arr1[3];
                cout << "], [" << arr1[2] << ", " << arr1[3];
                flag = true;
            }
            if (arr2.size() == 4) {
                if (flag) {
                    cout << "], [" << arr2[0] << ", " << arr2[1];
                    cout << "], [" << arr2[0] << ", " << arr2[2];
                    cout << "], [" << arr2[0] << ", " << arr2[3];
                    cout << "], [" << arr2[1] << ", " << arr2[2];
                    cout << "], [" << arr2[1] << ", " << arr2[3];
                    cout << "], [" << arr2[2] << ", " << arr2[3];
                } else {
                    cout << "[[" << arr2[0] << ", " << arr2[1];
                    cout << "], [" << arr2[0] << ", " << arr2[2];
                    cout << "], [" << arr2[0] << ", " << arr2[3];
                    cout << "], [" << arr2[1] << ", " << arr2[2];
                    cout << "], [" << arr2[1] << ", " << arr2[3];
                    cout << "], [" << arr2[2] << ", " << arr2[3];
                    flag = true;
                }
            }
            if (flag) {
                cout << "]]" << endl;
                break;
            }

            // 如果申请处理器个数为2，则选择同一链路剩余可用的处理器数量为3个的为次次佳
            if (arr1.size() == 3) {
                cout << "[[" << arr1[0] << ", " << arr1[1];
                cout << "], [" << arr1[0] << ", " << arr1[2];
                cout << "], [" << arr1[1] << ", " << arr1[2];
                flag = true;
            }
            if (arr2.size() == 3) {
                if (flag) {
                    cout << "], [" << arr2[0] << ", " << arr2[1];
                    cout << "], [" << arr2[0] << ", " << arr2[2];
                    cout << "], [" << arr2[1] << ", " << arr2[2];
                } else {
                    cout << "[[" << arr2[0] << ", " << arr2[1];
                    cout << "], [" << arr2[0] << ", " << arr2[2];
                    cout << "], [" << arr2[1] << ", " << arr2[2];
                    flag = true;
                }
            }
            if (flag) {
                cout << "]]" << endl;
                break;
            }
        }

        case 4: {
            bool flag = false;
            // 如果申请处理器个数为4，则选择同一链路剩余可用的处理器数量为4个的为最佳
            if (arr1.size() == 4) {
                cout << "[[" << arr1[0] << ", " << arr1[1] << ", " << arr1[2] << ", " << arr1[3];
                flag = true;
            }
            if (arr2.size() == 4) {
                if (flag) {
                    cout << "], [" << arr2[0] << ", " << arr2[1] << ", " << arr2[2] << ", " << arr2[3];
                } else {
                    cout << "[[" << arr2[0] << ", " << arr2[1] << ", " << arr2[2] << ", " << arr2[3];
                    flag = true;
                }
            }
            if (flag) {
                cout << "]]" << endl;
                break;
            }
        }

        case 8:
            // 如果申请处理器个数为8，则选择所有可用的处理器
            if (arr1.size() == 4 && arr2.size() == 4) {
                cout << "[[" << arr1[0] << ", " << arr1[1] << ", " << arr1[2] << ", " << arr1[3]
                    << ", " << arr2[0] << ", " << arr2[1] << ", " << arr2[2] << ", " << arr2[3] << "]]" << endl;
            }
            break;
    }

    return 0;
}
```
