### 自己

```c++
#include<iostream>
#include<vector>
#include<string>
#include<map>
#include<sstream>
#include<algorithm>

using namespace std;

bool compare(const pair<string, int>& lhs, const pair<string, int>& rhs) {
    if(lhs.second == rhs.second) {
        return lhs.first < rhs.first;
    }
    return lhs.second > rhs.second;
}

int main()
{
    int m;
    cin>>m;
    vector<int> weightVec(5, 0);
    for(int i = 0; i < 5; i++) {
        cin>>weightVec[i];
    }
    cin.ignore(); //忽略1个字符

    //key:   项目名称
    //value: 值
    map<string, int> projectMap;
    for(int i = 0; i < m; i++) {
        string line;
        getline(cin, line);
        stringstream ss(line);
        string key;
        int number;
        ss>>key;
        int sum = 0;
        vector<int> valueVec(5);
        int k = 0;
        while(ss>>number) {
            valueVec[k++] = number;
        }
        for(int j = 0; j < 5; j++) {
            sum += valueVec[j] * weightVec[j];
        }
        projectMap[key] = sum;
    }

    vector<pair<string, int>> sortProject(projectMap.begin(), projectMap.end());
    sort(sortProject.begin(), sortProject.end(), compare);
    for(auto subPair : sortProject) {
        cout<<subPair.first<<endl;
    }
    cout<<endl;
}
```
